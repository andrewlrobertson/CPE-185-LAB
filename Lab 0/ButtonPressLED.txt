void setup() {
  // put your setup code here, to run once:
  pinMode(11,OUTPUT);
  pinMode(8,INPUT);
  Serial.begin(9600);
}

void loop() {
  // put your main code here, to run repeatedly:
  if(digitalRead(8)==HIGH){
    Serial.println("Button pressed");
    digitalWrite(11,HIGH);
  }
  else{
    Serial.println("Button NOT pressed");
    digitalWrite(11,LOW);    
  }
  delay(10);
}
